## Task 5: 创建位置获取工具

**Files:**

- Create: `src/utils/location.ts`

**Step 1: 创建地理位置工具**

```typescript
export interface Location {
  latitude: number;
  longitude: number;
}

export async function getCurrentLocation(): Promise<Location> {
  return new Promise((resolve, reject) => {
    if (!navigator.geolocation) {
      reject(new Error("浏览器不支持地理位置"));
      return;
    }

    navigator.geolocation.getCurrentPosition(
      (position) => {
        resolve({
          latitude: position.coords.latitude,
          longitude: position.coords.longitude,
        });
      },
      (error) => {
        reject(error);
      },
      {
        enableHighAccuracy: true,
        timeout: 10000,
        maximumAge: 60000,
      },
    );
  });
}

export function buildMapUrl(
  platform: "dianping" | "baidu" | "amap" | "tencent",
  keyword: string,
  location?: Location,
): string {
  const encodedKeyword = encodeURIComponent(keyword);

  switch (platform) {
    case "dianping":
      return location
        ? `https://m.dianping.com/shoplist/1/search?from=m_search&keyword=${encodedKeyword}`
        : `https://www.dianping.com/search/keyword/203/0_${encodedKeyword}`;

    case "baidu":
      return `https://map.baidu.com/search/${encodedKeyword}/@${location?.longitude || 0},${location?.latitude || 0}`;

    case "amap":
      const aMapUrl = location
        ? `https://restapi.amap.com/v3/place/search?keywords=${encodedKeyword}&location=${location.longitude},${location.latitude}`
        : `https://www.amap.com/search?query=${encodedKeyword}`;
      return aMapUrl;

    case "tencent":
      return location
        ? `https://apis.map.qq.com/jsapi?qt=nearby&lon=${location.longitude}&lat=${location.latitude}&keywords=${encodedKeyword}`
        : `https://map.qq.com/search/${encodedKeyword}`;

    default:
      return "";
  }
}
```

<script setup lang="ts">
import { ref, computed, onMounted } from "vue";
import AnimatedBackground from "./components/AnimatedBackground.vue";
import FoodCard from "./components/FoodCard.vue";
import MapSelector from "./components/MapSelector.vue";
import { foodsData, getRandomFood, type FoodItem } from "./data/foods";
import {
  getCurrentLocation,
  buildMapUrl,
  type Location,
} from "./utils/location";

// 状态
const currentFood = ref<FoodItem | null>(null);
const isRandoming = ref(false);
const randomCount = ref(1);
const showMapSelector = ref(false);
const userLocation = ref<Location | null>(null);
const isVisible = ref(false);

// 按钮文案
const btnText = computed(() => {
  if (isRandoming.value) return "停止";
  if (randomCount.value === 1) return "开始";
  return "换一个";
});

// 切换随机/停止
function toggleRandom() {
  if (isRandoming.value) {
    stopRandom();
  } else {
    startRandom();
  }
}

let timer: number | null = null;

function startRandom() {
  if (randomCount.value >= 4) {
    alert("这么作，今天别吃了！");
    randomCount.value++;
    return;
  }

  randomCount.value++;
  isRandoming.value = true;
  isVisible.value = false;

  timer = window.setInterval(() => {
    currentFood.value = getRandomFood();
  }, 80);
}

function stopRandom() {
  if (timer) {
    clearInterval(timer);
    timer = null;
  }
  isRandoming.value = false;

  // 确保最终选中有食物
  if (!currentFood.value) {
    currentFood.value = getRandomFood();
  }

  // 触发动画
  setTimeout(() => {
    isVisible.value = true;
  }, 100);
}

// 打开地图选择器
function openMapSelector() {
  showMapSelector.value = true;
}

// 获取位置并跳转
async function handleMapSelect(platform: string) {
  if (!currentFood.value) return;

  showMapSelector.value = false;

  try {
    if (!userLocation.value) {
      userLocation.value = await getCurrentLocation();
    }
    const url = buildMapUrl(
      platform as any,
      currentFood.value.name,
      userLocation.value,
    );
    window.open(url, "_blank");
  } catch (error) {
    // 如果获取位置失败，使用不带位置的 URL
    const url = buildMapUrl(platform as any, currentFood.value.name);
    window.open(url, "_blank");
  }
}

// 初始化
onMounted(() => {
  currentFood.value = getRandomFood();
  // 初始显示
  setTimeout(() => {
    isVisible.value = true;
  }, 500);
});
</script>

<template>
  <div class="app">
    <AnimatedBackground />

    <main class="main-content">
      <FoodCard
        v-if="currentFood"
        :food-name="currentFood.name"
        :food-image="currentFood.image"
        :is-visible="isVisible"
      />

      <button
        class="action-btn"
        :class="{ 'is-randoming': isRandoming }"
        @click="toggleRandom"
      >
        {{ btnText }}
      </button>

      <button
        v-if="currentFood && !isRandoming && randomCount > 1"
        class="search-btn"
        @click="openMapSelector"
      >
        <span>去附近找找</span>

          <circle cx="12" cy="10" r="3" />
        </svg>
      </button>
    </main>

    <MapSelector
      :is-visible="showMapSelector"
      @close="showMapSelector = false"
      @select="handleMapSelect"
    />
  </div>
</template>
